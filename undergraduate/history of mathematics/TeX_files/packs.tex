\newcommand{\thmboxstyle}[4]{
	\mdfdefinestyle{#2}{
		linecolor = #3,
		backgroundcolor = #4,
		nobreak = true
	}
	\declaretheoremstyle[
	headfont = \bfseries\color{#3},
	mdframed = {style = #2},
	headpunct = {\\[0.4pt]},
	postheadspace = {0pt},
	]{#1}
}

			\hfill --- #2
		\end{minipage}
	\end{flushright}
